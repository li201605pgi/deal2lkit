35 template <int spacedim, int n_components>
36 void ParsedMappedFunctions<spacedim,n_components>::add_normal_components()
37 {
38  std::vector<std::string> var = unique(_component_names);
39  for (unsigned int i=0; i<var.size(); ++i)
40  {
41  int n = std::count(_component_names.begin(),_component_names.end(), var[i]);
42  if (n>1)
43  {
44  _normal_components.push_back(var[i]+".N");

46  std::pair<unsigned int, std::string> nc (pos, var[i]+".N");
47  normal_components.push_back(nc);
48  }
49  }

52 template <int spacedim, int n_components>
53 void ParsedMappedFunctions<spacedim,n_components>::parse_parameters_call_back()
54 {
55  add_normal_components();
56  _all_components = _component_names;
57  for (unsigned int i=0; i<_normal_components.size(); ++i)
58  {
59  _all_components.push_back(_normal_components[i]);
60  }
61  split_id_components(str_id_components);
62  split_id_functions(str_id_functions,str_constants);
63  AssertThrow(id_components.size() == id_functions.size(),
64  ExcIdsMismatch(id_components.size(), id_functions.size()));

68  {
69  std::pair<ComponentMask, shared_ptr<dealii::Functions::ParsedFunction<spacedim> > > mapped(it->second, id_functions[it->first]);
70 
71  mapped_functions[it->first] = mapped;
72 
73  }
74 
75  set_normal_functions();

79 template <int spacedim, int n_components>
80 void ParsedMappedFunctions<spacedim,n_components>::split_id_components(const std::string &parsed_idcomponents)
81 {
82  std::vector<std::string> idcomponents;
83 
84  idcomponents = Utilities::split_string_list(parsed_idcomponents, '%');
85 
86  for (unsigned int i=0; i<idcomponents.size(); ++i)
87  {
88  std::vector<std::string> id_comp;
89  std::vector<std::string> components;
90  std::vector<bool> mask(n_components,false);
91 
92  id_comp = Utilities::split_string_list(idcomponents[i], '=');
93 
94  unsigned int id = Utilities::string_to_int(id_comp[0]);
95  ids.push_back(id);
96 
97  components = Utilities::split_string_list(id_comp[1], ';');
98  AssertThrow(components.size() <= n_components,
99  ExcMessage("Wrong number of components specified for id " + id_comp[0]));
100  for (unsigned int c=0; c<components.size(); ++c)
101  {
102  if (components[c] == "ALL")
103  {
104  for (unsigned int j=0; j<n_components; ++j)
105  mask[j] = true;
106  break;
107  }
108 
109  AssertThrow(_component_names.size() == n_components,
110  ExcMessage("Parsed components name must match the number of components of the problem"));
111 
112  if ((std::find(_component_names.begin(), _component_names.end(), components[c]) != _component_names.end()))
113  for (unsigned int j=0; j<_component_names.size(); ++j)
114  mask[j] = (_component_names[j] == components[c] || mask[j]);
115 
116  else if ((std::find(_normal_components.begin(), _normal_components.end(), components[c]) != _normal_components.end()))
117  {
118  normal_ids.push_back(id);
119  for (unsigned int j=0; j<normal_components.size(); ++j)
120  if (normal_components[j].second == components[c])
121  {
122  mapped_normal_components[components[c]].second = normal_components[j].first;
123  mapped_normal_components[components[c]].first.push_back(id);
124  }
125  }
126  else
127  {
128  try
129  {
130  unsigned int m = Utilities::string_to_int(components[c]);
131  AssertThrow(m < n_components, ExcWrongComponent(id,m,n_components));
132  mask[m] = true;
133  }
134  catch (std::exception &exc)
135  AssertThrow(false, ExcWrongVariable(id,components[c],_all_components));
136  }
137 
138  }
139  id_components[id] = ComponentMask(mask);
140  }
141 }

143 template <int spacedim, int n_components>
144 void ParsedMappedFunctions<spacedim,n_components>::split_id_functions(const std::string &parsed_idfunctions,
145  const std::string &constants)
146 {
147  std::vector<unsigned int> id_defined_functions;
148 
149  // if it is empty a ZeroFunction<dim>(n_components) is applied on the
150  // parsed ids in the components
151  if (parsed_idfunctions == "")
152  {
153  for (unsigned int i=0; i<ids.size(); ++i)
154  {
155  id_defined_functions.push_back(ids[i]);
156  shared_ptr<dealii::Functions::ParsedFunction<spacedim> > ptr;
157 
158  ParameterHandler internal_prm;
159  dealii::Functions::ParsedFunction<spacedim>::declare_parameters(internal_prm, n_components);
160  ptr = SP(new dealii::Functions::ParsedFunction<spacedim>(n_components));
161  ptr->parse_parameters(internal_prm);
162 
163  id_functions[ids[i]] = ptr;
164  std::string str;
165  for (unsigned int j=0; j<n_components-1; ++j)
166  str += "0;";
167  str+="0";
168 
169  id_str_functions[ids[i]] = str;
170 
171  }
172  }
173  else
174  {
175 
176 
177  std::vector<std::string> idfunctions;
178  idfunctions = Utilities::split_string_list(parsed_idfunctions, '%');
179 
180  for (unsigned int i=0; i<idfunctions.size(); ++i)
181  {
182  std::vector<std::string> id_func;
183 
184  id_func = Utilities::split_string_list(idfunctions[i], '=');
185 
186  unsigned int id = Utilities::string_to_int(id_func[0]);
187  id_str_functions[id] = id_func[1];
188 
189  // check if the current id is also defined in id_components
190  AssertThrow((std::find(ids.begin(), ids.end(), id) != ids.end()),
191  ExcIdNotMatch(id));
192  id_defined_functions.push_back(id);
193 
194  shared_ptr<dealii::Functions::ParsedFunction<spacedim> > ptr;
195 
196  ParameterHandler internal_prm;
197  dealii::Functions::ParsedFunction<spacedim>::declare_parameters(internal_prm, n_components);
198  internal_prm.set("Function expression", id_func[1]);
199  internal_prm.set("Function constants", constants);
200  ptr = SP(new dealii::Functions::ParsedFunction<spacedim>(n_components));
201  ptr->parse_parameters(internal_prm);
202 
203  id_functions[id] = ptr;
204 
205 
206  }
207  }
208 
209  // check if the number of ids defined in id_components and id_functions are the same
210  AssertThrow(ids.size() == id_defined_functions.size(),
211  ExcMessage("Ids associated to components and to functions are not the same."));
212 
213 }

317 template <int spacedim, int n_components>
318 void ParsedMappedFunctions<spacedim,n_components>::set_normal_functions()
319 {
320  typedef std::map<std::string, std::pair<std::vector<unsigned int>, unsigned int > >::iterator it_type;
321 
322  for (it_type it=mapped_normal_components.begin(); it != mapped_normal_components.end(); ++it)
323  {
324 
325  std::vector<unsigned int> normal_ids = (it->second).first;
326  unsigned int fcv = (it->second).second; // unsigned int first component vector
327 
328  for (unsigned int i=0; i<normal_ids.size(); ++i)
329  {
330  std::vector<std::string> normal_func;
331  normal_func = Utilities::split_string_list(id_str_functions[normal_ids[i]], ';');
332  shared_ptr<dealii::Functions::ParsedFunction<spacedim> > normal_ptr;
333 
334  ParameterHandler normal_prm;
335  dealii::Functions::ParsedFunction<spacedim>::declare_parameters(normal_prm, spacedim);
336  std::string str_normal_func;
337  Assert(spacedim>1, ExcNotImplemented());
338  for (unsigned int j=0; j<spacedim-1; ++j)
339  str_normal_func += normal_func[fcv+j] + ";";
340  str_normal_func += normal_func[fcv+spacedim-1];
341 
342  normal_prm.set("Function expression", str_normal_func);
343  normal_prm.set("Function constants", str_constants);
344  normal_ptr = SP(new dealii::Functions::ParsedFunction<spacedim>(spacedim));
345  normal_ptr->parse_parameters(normal_prm);
346  std::pair<unsigned int, unsigned int> id_fcv (normal_ids[i],fcv);
347  _normal_functions[id_fcv]=normal_ptr;
348  }
349  }
