300  void clear(bool force=false)
301  {
302  if (force)
303  {
304  while (current_line < n_lines-1)
305  {
306  stream_out << "\033[B";
307  current_line++;
308  };
309 
310  if (current_line>0)
311  {
312  for (unsigned int i=0; i<n_lines; ++i)
313  {
314  stream_out << "\r" << std::setfill(' ') << std::setw(width) << " " << "\r" << "\033[A";
315  current_line--;
316  if (current_line==0)
317  break;
318  }
319  stream_out << "\r" << std::setfill(' ') << std::setw(width) << " " << "\r";
320  }
321  else
322  {
323  stream_out << "\r" << std::setfill(' ') << std::setw(width) << " " << "\r";
324  }
325  clear_next = false;
326  }
327  else if (clear_next)
328  {
329  stream_out << "\r" << std::setfill(' ') << std::setw(width) << " " << "\r";
330  stream_out << "\033[A" << std::endl;
331  clear_next = false;
332  }
