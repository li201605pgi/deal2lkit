334  template<int dim, int spacedim>
335  void
336  default_create_grid( Triangulation<dim,spacedim> &tria,
337  std::string grid_name,
338  Point<spacedim> point_option_one,
339  Point<spacedim> point_option_two,
340  double double_option_two,
341  double double_option_one,
342  double double_option_three,
343  bool colorize,
344  unsigned int un_int_option_one,
345  unsigned int un_int_option_two,
346  std::vector<unsigned int> un_int_vec_option_one,
347  std::string input_grid_file_name)
348  {
349  if (grid_name == "rectangle" || grid_name == "subdivided_hyper_rectangle")
350  {
351  GridGenerator::subdivided_hyper_rectangle (tria,
352  un_int_vec_option_one,
353  point_option_two,
354  point_option_one,
355  colorize);
356  }
357  else if (grid_name == "file")
358  {
359  GridIn<dim, spacedim> gi;
360  gi.attach_triangulation(tria);
361 
362  std::ifstream in(input_grid_file_name.c_str());
363  AssertThrow(in, ExcIO());
364 
365  std::string ext = extension(input_grid_file_name);
366  if (ext == "vtk")
367  gi.read_vtk(in);
368  else if (ext == "msh")
369  gi.read_msh(in);
370  else if (ext == "ucd" || ext == "inp")
371  gi.read_ucd(in);
372  else if (ext == "unv")
373  gi.read_unv(in);
374  else
375  Assert(false, ExcNotImplemented());
376  }
377  // silence warnings of unused variables
378  (void)double_option_one;
379  (void)double_option_two;
380  (void)double_option_three;
381  (void)un_int_option_one;
382  (void)un_int_option_two;
383  }

704 typename Triangulation<dim,spacedim>::MeshSmoothing
705 ParsedGridGenerator<dim, spacedim>::get_smoothing()
706 {
707  int smoothing = 0;
708  std::vector<std::string> ss = Utilities::split_string_list(mesh_smoothing);
709  for (std::string s : ss)
710  {
711  if (s == "none")
712  smoothing = smoothing|(int)Triangulation<dim,spacedim>::none;
713  else if (s == "limit_level_difference_at_vertices")
714  smoothing = smoothing|(int)Triangulation<dim,spacedim>::limit_level_difference_at_vertices;
715  else if (s == "eliminate_unrefined_islands")
716  smoothing = smoothing|(int)Triangulation<dim,spacedim>::eliminate_unrefined_islands;
717  else if (s == "patch_level_1")
718  smoothing = smoothing|(int)Triangulation<dim,spacedim>::patch_level_1;
719  else if (s == "coarsest_level_1")
720  smoothing = smoothing|(int)Triangulation<dim,spacedim>::coarsest_level_1;
721  else if (s == "allow_anisotropic_smoothing")
722  smoothing = smoothing|(int)Triangulation<dim,spacedim>::allow_anisotropic_smoothing;
723  else if (s == "eliminate_refined_inner_islands")
724  smoothing = smoothing|(int)Triangulation<dim,spacedim>::eliminate_refined_inner_islands;
725  else if (s == "eliminate_refined_boundary_islands")
726  smoothing = smoothing|(int)Triangulation<dim,spacedim>::eliminate_refined_boundary_islands;
727  else if (s == "do_not_produce_unrefined_islands")
728  smoothing = smoothing|(int)Triangulation<dim,spacedim>::do_not_produce_unrefined_islands;
729  else if (s == "smoothing_on_refinement")
730  smoothing = smoothing|(int)Triangulation<dim,spacedim>::smoothing_on_refinement;
731  else if (s == "smoothing_on_coarsening")
732  smoothing = smoothing|(int)Triangulation<dim,spacedim>::smoothing_on_coarsening;
733  else if (s == "maximum_smoothing")
734  smoothing = smoothing|(int)Triangulation<dim,spacedim>::maximum_smoothing;
735  else
736  Assert(false, ExcInternalError("Found a non supported smoothing. Should not happen:"+s))
737  }
