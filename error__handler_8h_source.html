299 template <int ntables>
300 template<typename DH, typename VECTOR>
301 void ErrorHandler<ntables>::error_from_exact(const Mapping<DH::dimension, DH::space_dimension> &mapping,
302  const DH &dh,
303  const VECTOR &solution,
304  const Function<DH::space_dimension> &exact,
305  unsigned int table_no,
306  double dt)
307 {
308  const int dim=DH::dimension;
309  const int spacedim=DH::space_dimension;
310  if (compute_error)
311  {
312  AssertThrow(initialized, ExcNotInitialized());
313  AssertThrow(table_no < types.size(), ExcIndexRange(table_no, 0, names.size()));
314  AssertThrow(exact.n_components == types[table_no].size(),
315  ExcDimensionMismatch(exact.n_components, types[table_no].size()));
316 
317  std::vector< std::vector<double> > error( exact.n_components, std::vector<double>(4));
318  const unsigned int n_active_cells = dh.get_tria().n_global_active_cells();
319  const unsigned int n_dofs=dh.n_dofs();
320 
321  if (extras[table_no]["cells"])
322  {
323  tables[table_no].add_value("cells", n_active_cells);
324  tables[table_no].set_tex_caption("cells", "\\# cells");
325  tables[table_no].set_tex_format("cells", "r");
326  }
327  if (extras[table_no]["dofs"])
328  {
329  tables[table_no].add_value("dofs", n_dofs);
330  tables[table_no].set_tex_caption("dofs", "\\# dofs");
331  tables[table_no].set_tex_format("dofs", "r");
332  }
333  if (extras[table_no]["dt"])
334  {
335  tables[table_no].add_value("dt", dt);
336  tables[table_no].set_tex_caption("dt", "\\Delta t");
337  tables[table_no].set_tex_format("dt", "r");
338  }
339 
340  bool compute_Linfty = false;
341  bool compute_L2 = false;
342  bool compute_W1infty = false;
343  bool compute_H1 = false;
344  bool add_this = false;
345 
346  unsigned int last_non_add = 0;
347 
348  for (unsigned int component=0; component < exact.n_components; ++component)
349  {
350  NormFlags norm = types[table_no][component];
351 
352  // Select one Component
353  ComponentSelectFunction<spacedim> select_component ( component, 1. , exact.n_components);
354 
355  Vector<float> difference_per_cell (dh.get_tria().n_global_active_cells());
356 
357  QGauss<dim> q_gauss((dh.get_fe().degree+1) * 2);
358 
359  // The add bit is set
360  add_this = (norm & AddUp);
361 
362  if (!add_this)
363  {
364  last_non_add = component;
365  compute_L2 = ( norm & L2 );
366  compute_H1 = ( norm & H1 );
367  compute_W1infty = ( norm & W1infty ) ;
368  compute_Linfty = ( norm & Linfty );
369  }
370  // if add is set, we do not modify the previous selection
371 
372  if (compute_L2)
373  {
374  VectorTools::integrate_difference (
375  mapping,
376  dh,
377  solution,
378  exact,
379  difference_per_cell,
380  q_gauss,
381  VectorTools::L2_norm,
382  &select_component );
383  }
384 
385  const double L2_error = difference_per_cell.l2_norm();
386  difference_per_cell = 0;
387 
388  if (compute_H1)
389  {
390  VectorTools::integrate_difference (
391  mapping,
392  dh, //dof_handler,
393  solution,
394  exact,
395  difference_per_cell,
396  q_gauss,
397  VectorTools::H1_norm,
398  &select_component );
399  }
400  const double H1_error = difference_per_cell.l2_norm();
401  difference_per_cell = 0;
402 
403  if (compute_W1infty)
404  {
405  VectorTools::integrate_difference (
406  mapping,
407  dh, //dof_handler,
408  solution,
409  exact,
410  difference_per_cell,
411  q_gauss,
412  VectorTools::W1infty_norm,
413  &select_component );
414  }
415 
416  const double W1inf_error = difference_per_cell.linfty_norm();
417 
418  if (compute_Linfty)
419  {
420  VectorTools::integrate_difference (
421  mapping,
422  dh, //dof_handler,
423  solution,
424  exact,
425  difference_per_cell,
426  q_gauss,
427  VectorTools::Linfty_norm,
428  &select_component );
429  }
430 
431  const double Linf_error = difference_per_cell.linfty_norm();
432 
433  if (add_this)
434  {
435  AssertThrow(component, ExcMessage("Cannot add on first component!"));
436 
437  error[last_non_add][0] = std::max(error[last_non_add][0], Linf_error);
438  error[last_non_add][1] += L2_error;
439  error[last_non_add][2] = std::max(error[last_non_add][2], W1inf_error);
440  error[last_non_add][3] += H1_error;
441 
442  }
443  else
444  {
445 
446  error[component][0] = Linf_error;
447  error[component][1] = L2_error;
448  error[component][2] = W1inf_error;
449  error[component][3] = H1_error;
450 
451  }
452  }
453 
454  for (unsigned int j=0; j<exact.n_components; ++j)
455  {
456  NormFlags norm = types[table_no][j];
457  // If this was added, don't do anything
458  if ( !(norm & AddUp) )
459  {
460  if (norm & Linfty)
461  {
462  std::string name = headers[j] + "_Linfty";
463  std::string latex_name = "$\\| " +
464  latex_headers[j] + " - " +
465  latex_headers[j] +"_h \\|_\\infty $";
466  double this_error = error[j][0];
467 
468  tables[table_no].add_value(name, this_error);
469  tables[table_no].set_precision(name, 3);
470  tables[table_no].set_scientific(name, true);
471  tables[table_no].set_tex_caption(name, latex_name);
472  }
473 
474  if (norm & L2)
475  {
476  std::string name = headers[j] + "_L2";
477  std::string latex_name = "$\\| " +
478  latex_headers[j] + " - " +
479  latex_headers[j] +"_h \\|_0 $";
480  double this_error = error[j][1];
481 
482  tables[table_no].add_value(name, this_error);
483  tables[table_no].set_precision(name, 3);
484  tables[table_no].set_scientific(name, true);
485  tables[table_no].set_tex_caption(name, latex_name);
486  }
487  if (norm & W1infty)
488  {
489  std::string name = headers[j] + "_W1infty";
490  std::string latex_name = "$\\| " +
491  latex_headers[j] + " - " +
492  latex_headers[j] +"_h \\|_{1,\\infty} $";
493  double this_error = error[j][2];
494 
495  tables[table_no].add_value(name, this_error);
496  tables[table_no].set_precision(name, 3);
497  tables[table_no].set_scientific(name, true);
498  tables[table_no].set_tex_caption(name, latex_name);
499  }
500  if (norm & H1)
501  {
502  std::string name = headers[j] + "_H1";
503  std::string latex_name = "$\\| " +
504  latex_headers[j] + " - " +
505  latex_headers[j] +"_h \\|_1 $";
506  double this_error = error[j][3];
507 
508  tables[table_no].add_value(name, this_error);
509  tables[table_no].set_precision(name, 3);
510  tables[table_no].set_scientific(name, true);
511  tables[table_no].set_tex_caption(name, latex_name);
512  }
513  }
514  }
515  }
