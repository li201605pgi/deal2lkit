137 template<typename VECTOR>
138 void ParsedDataOut<dim,spacedim>::add_data_vector(const VECTOR &data_vector,
139  const std::string &desc)
140 {
141  AssertThrow(initialized, ExcNotInitialized());
142  deallog.push("AddingData");
143  std::vector<std::string> dd = Utilities::split_string_list(desc);
144  if (data_out.default_suffix() != "")
145  {
146  if (dd.size() ==1 )
147  {
148  data_out.add_data_vector (data_vector, desc);
149  }
150  else
151  {
152  std::vector<std::string>::iterator sit = dd.begin();
153  std::vector<int> occurrances;
154 
155  for (; sit!=dd.end(); ++sit)
156  occurrances.push_back(std::count (dd.begin(), dd.end(), *sit));
157 
158  std::vector<int>::iterator iit = occurrances.begin();
159  sit = dd.begin();
160 
161  std::vector<DataComponentInterpretation::DataComponentInterpretation>
162  data_component_interpretation;
163 
164  for (; iit !=occurrances.end(); ++iit, ++sit)
165  {
166  if (*iit > 1)
167  data_component_interpretation.push_back(DataComponentInterpretation::component_is_part_of_vector);
168  else
169  data_component_interpretation.push_back(DataComponentInterpretation::component_is_scalar);
170  }
171 
172  data_out.add_data_vector (data_vector, dd, DataOut<dim, DoFHandler<dim, spacedim> >::type_dof_data,
173  data_component_interpretation);
174 
175  }
176  deallog << "Added data: " << desc << std::endl;
177  }
178  deallog.pop();
179 }
