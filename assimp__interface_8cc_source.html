52  template <int dim, int spacedim>
53  bool generate_triangulation(const std::string filename,
54  Triangulation<dim,spacedim> &tria,
55  int mesh_index,
56  bool remove_duplicates,
57  double tol)
58  {
59  // Only good for surface grids.
60  AssertThrow(dim<3, ExcImpossibleInDim(dim));
61 
62  // Create an istance of the Importer class
63  Assimp::Importer importer;
64 
65  // And have it read the given file with some postprocessing
66  const aiScene *scene = importer.ReadFile( filename.c_str(),
67  aiProcess_RemoveComponent |
68  aiProcess_JoinIdenticalVertices |
69  aiProcess_ImproveCacheLocality |
70  aiProcess_SortByPType |
71  aiProcess_OptimizeGraph |
72  aiProcess_OptimizeMeshes);
73 
74  // If the import failed, report it
75  if ( !scene)
76  {
77  cerr << importer.GetErrorString() << endl;
78  return false;
79  }
80 
81  if ( scene->mNumMeshes == 0)
82  {
83  cout << "Input file contains no meshes." << endl;
84  return false;
85  }
86 
87  AssertThrow((mesh_index == -1) || (mesh_index < (int) scene->mNumMeshes),
88  ExcMessage("Too few meshes in the file."));
89 
90  cout << "The given file contains "
91  << scene->mNumMeshes << " meshes. ";
92  if (mesh_index == -1)
93  cout << "Importing all available meshes." << endl;
94  else
95  cout << "Trying to import mesh number " << mesh_index << endl;
96 
97  int start_mesh = (mesh_index == -1 ? 0 : mesh_index);
98  int end_mesh = (mesh_index == -1 ? (int) scene->mNumMeshes : mesh_index+1);
99 
100 
101 
102  // Deal.II objects are created empty, and then filled with imported file.
103  vector<Point<spacedim> > vertices;
104  vector<CellData<dim> > cells;
105  SubCellData subcelldata;
106 
107  // A series of counters to merge cells.
108  unsigned int v_offset=0;
109  unsigned int c_offset=0;
110 
111  // The index of the mesh will be used as a material index.
112  for (int m=start_mesh; m<end_mesh; ++m)
113  {
114 
115  const aiMesh *mesh = scene->mMeshes[m];
116 
117  // Check that we know what to do with this mesh, otherwise just
118  // ignore it
119  if ( (dim == 2) && mesh->mPrimitiveTypes != aiPrimitiveType_POLYGON)
120  {
121  cout << "Skipping incompatible mesh " << m
122  << "/" << scene->mNumMeshes << "." << endl;
123  continue;
124  }
125  else if ( (dim == 1) && mesh->mPrimitiveTypes != aiPrimitiveType_LINE)
126  {
127  cout << "Skipping incompatible mesh " << m
128  << "/" << scene->mNumMeshes << "." << endl;
129  continue;
130  }
131  // Vertices
132  const unsigned int n_vertices = mesh->mNumVertices;
133  const aiVector3D *mVertices = mesh->mVertices;
134 
135  // Faces
136  const unsigned int n_faces = mesh->mNumFaces;
137  const aiFace *mFaces = mesh->mFaces;
138 
139  vertices.resize(v_offset+n_vertices);
140  cells.resize(c_offset+n_faces);
141 
142  cout << "Input mesh has " << n_vertices << " vertices and "
143  << n_faces << " faces." << endl;
144 
145 
146  for (unsigned int i=0; i<n_vertices; ++i)
147  for (unsigned int d=0; d<spacedim; ++d)
148  vertices[i+v_offset][d] = mVertices[i][d];
149 
150  unsigned int valid_cell = c_offset;
151  for (unsigned int i=0; i<n_faces; ++i)
152  {
153  if (mFaces[i].mNumIndices == GeometryInfo<dim>::vertices_per_cell)
154  {
155  cout << "Face: " << i << ": ";
156  for (unsigned int f=0; f<GeometryInfo<dim>::vertices_per_cell; ++f)
157  {
158  cells[valid_cell].vertices[f] = mFaces[i].mIndices[f]+v_offset;
159  cout << cells[valid_cell].vertices[f] << " ";
160  }
161  cout << endl;
162  cells[valid_cell].material_id = (types::material_id) m;
163  ++valid_cell;
164  }
165  else
166  {
167  cout << "Skipping face " << i << " of mesh "
168  << m << ", it has "
169  << mFaces[i].mNumIndices << " vertices. We expect "
170  << GeometryInfo<dim>::vertices_per_cell << endl;
171  }
172  }
173  cells.resize(valid_cell);
174 
175  // The vertices are added all at once. Cells are check for
176  // validity, so only valid_cells are now present in the deal.II
177  // list of cells.
178  v_offset += n_vertices;
179  c_offset = valid_cell;
180  }
181 
182  // No cells were read
183  if (cells.size() == 0)
184  return false;
185 
186  if (remove_duplicates)
187  {
188  vector<unsigned int> considered_vertices;
189  GridTools::delete_duplicated_vertices(vertices, cells, subcelldata,
190  considered_vertices, tol);
191 
192  GridTools::delete_duplicated_vertices(vertices, cells, subcelldata,
193  considered_vertices, tol);
194 
195  GridTools::delete_duplicated_vertices(vertices, cells, subcelldata,
196  considered_vertices, tol);
197 
198  GridTools::delete_duplicated_vertices(vertices, cells, subcelldata,
199  considered_vertices, tol);
200 
201  GridTools::delete_duplicated_vertices(vertices, cells, subcelldata,
202  considered_vertices, tol);
203  }
204  GridTools::delete_unused_vertices(vertices, cells, subcelldata);
205  if (dim == spacedim)
206  GridReordering<dim, spacedim>::invert_all_cells_of_negative_grid(vertices,
207  cells);
208 
209  cout << "Creating tria with " << vertices.size() << " vertices, "
210  << cells.size() << "cells. " << std::endl;
211 
212  GridReordering<dim, spacedim>::reorder_cells(cells);
213  if (dim == 2)
214  tria.create_triangulation_compatibility(vertices, cells, subcelldata);
215  else
216  tria.create_triangulation(vertices, cells, subcelldata);
217 
218  cout << "Generated mesh has " << tria.n_vertices() << " vertices and "
219  << tria.n_active_cells() << " active cells. " << endl;
220 
221  typename Triangulation<dim,spacedim>::active_cell_iterator cell = tria.begin_active(),
222  endc=tria.end();
223 
224  unsigned int boundary_faces = 0;
225  for (; cell != endc; ++cell)
226  if (cell->at_boundary())
227  boundary_faces++;
228 
229  cout << "Triangulation has " << boundary_faces << " boundary cells." << endl;
230  return true;
231 
232  }
