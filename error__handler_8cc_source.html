78  prm.declare_entry ("Solution names for latex", solution_names, Patterns::Anything(),
79  "Comma separated version of the same thing as above for "
80  "the latex version of the table.");
81 
82  // prm.declare_entry ("Ib output format", "msh", Patterns::Selection("raw|msh"));
83  // prm.declare_entry ("Ib input file prefix", "ellipse", Patterns::Anything());
84  for (unsigned int i=0; i<ntables; ++i)
85  {
86  char tmp[10];
87  sprintf(tmp, "Table %d", i);

112 template <int ntables>
113 void ErrorHandler<ntables>::parse_parameters (ParameterHandler &prm)
114 {
115  write_error = prm.get_bool ("Write error files");
116  output_error = prm.get_bool ("Output error tables");
117 
118  error_file_format = prm.get ("Error file format");
119  compute_error = prm.get_bool ("Compute error");
120  std::string all_names = prm.get ("Table names");
121  headers = Utilities::split_string_list(prm.get ("Solution names"));
122  latex_headers = Utilities::split_string_list(prm.get ("Solution names for latex"));
123 
124  if (all_names != "")
125  {
126  names = Utilities::split_string_list(all_names);
127  Assert(names.size() <= ntables,
128  ExcMessage("You tried to construct more tables than you have compiled for."));
129  types.resize(names.size(), std::vector<NormFlags> (headers.size()));
130  add_rates.resize(names.size());
131  tables.resize(names.size());
132  latex_captions.resize(names.size());
133  std::map<std::string, bool> extra;
134  extra["dof"] = false;
135  extra["cells"] = false;
136  extra["dt"] = false;
137 
138  extras.resize(names.size(), extra);
139  rate_keys.resize(names.size(), "");
140 
141  for (unsigned int i=0; i<names.size(); ++i)
142  {
143  char tmp[10];
144  sprintf(tmp, "Table %d", i);
145  prm.enter_subsection(tmp);
146 
147  all_names = prm.get("List of error norms to compute");
148  add_rates[i] = prm.get_bool("Add convergence rates");
149  rate_keys[i] = prm.get("Rate key");
150  latex_captions[i] = prm.get("Latex table caption");
151  std::vector<std::string> all_extras =
152  Utilities::split_string_list(prm.get("Extra terms"));
153 
154  for (unsigned int x=0; x< all_extras.size(); ++x)
155  extras[i][all_extras[x]] = true;
156 
157  prm.leave_subsection();
158 
159  std::vector<std::string> all_comps = Utilities::split_string_list(all_names, ';');
160  // Check that the input string has all the needed fields
161  AssertThrow(all_comps.size() == headers.size(),
162  ExcDimensionMismatch(all_comps.size() , headers.size()));
163 
164  for (unsigned int j=0; j<all_comps.size(); ++j)
165  {
166  std::vector<std::string> all_types =
167  Utilities::split_string_list(all_comps[j]);
168  // If we named it the same, defaults to AddUp
169  if (j>0 && headers[j] == headers[j-1])
170  {
171  types[i][j] |= AddUp;
172  continue;
173  }
174  for (unsigned int k=0; k<all_types.size(); ++k)
175  {
176  if (all_types[k] == "Linfty")
177  {
178  types[i][j] |= Linfty;
179  }
180  else if (all_types[k] == "L2")
181  {
182  types[i][j] |= L2;
183  }
184  else if (all_types[k] == "W1infty")
185  {
186  types[i][j] |= W1infty;
187  }
188  else if (all_types[k] == "H1")
189  {
190  types[i][j] |= H1;
191  }
192  else if (all_types[k] == "AddUp")
193  {
194  types[i][j] |= AddUp;
195  }
196  else if (all_types[k] == "Custom")
197  {
198  types[i][j] |= Custom;
199  }
200  else
201  {
202  AssertThrow(false, ExcMessage("Didn't recognize a norm type."));
203  }
204  }
205  }
206  }
207  }
208  initialized = true;
209 }

211 template <int ntables>
212 void ErrorHandler<ntables>::output_table (std::ostream &out, const unsigned int table_no)
213 {
214  if (compute_error)
215  {
216  AssertThrow(initialized, ExcNotInitialized());
217  AssertThrow(table_no < names.size(), ExcIndexRange(table_no, 0, names.size()));
218 
219  // Add convergence rates
220  if (add_rates[table_no])
221  {
222  if (extras[table_no]["dofs"])
223  tables[table_no].omit_column_from_convergence_rate_evaluation("dofs");
224  if (extras[table_no]["cells"])
225  tables[table_no].omit_column_from_convergence_rate_evaluation("cells");
226  if (extras[table_no]["dt"])
227  tables[table_no].omit_column_from_convergence_rate_evaluation("dt");
228  if (rate_keys[table_no] == "")
229  tables[table_no].evaluate_all_convergence_rates(ConvergenceTable::reduction_rate_log2);
230  else
231  tables[table_no].evaluate_all_convergence_rates(rate_keys[table_no], ConvergenceTable::reduction_rate_log2);
232  }
233 
234  if (output_error) tables[table_no].write_text(out);
235 
236  if (write_error)
237  {
238  std::string filename = names[table_no] +
239  "." + error_file_format;
240 
241  std::ofstream table_file(filename.c_str());
242 
243  if (error_file_format == "tex")
244  tables[table_no].write_tex(table_file);
245  else if (error_file_format == "txt")
246  tables[table_no].write_text(table_file);
247  else if (error_file_format == "gpl")
248  tables[table_no].write_text(table_file,
249  TableHandler::table_with_separate_column_description);
250  else if (error_file_format == "org")
251  tables[table_no].write_text(table_file,
252  TableHandler::org_mode_table);
253  table_file.close();
254  }
255  }
256 }
