96 template<int dim, int spacedim>
97 void ParsedFiniteElement<dim,spacedim>::parse_parameters_call_back()
98 {
99  component_blocks.resize(component_names.size());
100  block_names.resize(component_names.size());
101  unsigned int j=0;
102  for (unsigned int i=0; i<component_names.size(); ++i)
103  {
104  if ( (i>0) && (component_names[i-1] != component_names[i]) )
105  j++;
106  component_blocks[i] = j;
107  block_names[j] = component_names[i];
108  }
109  block_names.resize(j+1);
110  FiniteElement<dim,spacedim> *fe = (*this)();
111  const unsigned int nc = fe->n_components();
112  delete fe;
113  AssertThrow(component_names.size() == nc,
114  ExcInternalError("Generated FE has the wrong number of components."));
115 
116  coupling = to_coupling(coupling_int);
117  preconditioner_coupling = to_coupling(preconditioner_coupling_int);
118 }

122 template<int dim, int spacedim>
123 Table<2,DoFTools::Coupling> ParsedFiniteElement<dim,spacedim>::to_coupling(const std::vector<std::vector<unsigned int> > &coupling_table) const
124 {
125  const unsigned int nc = n_components();
126  const unsigned int nb = n_blocks();
127 
128  Table<2,DoFTools::Coupling> out_coupling(nc, nc);
129 
130  std::vector<DoFTools::Coupling> m(3);
131  m[0] = DoFTools::none;
132  m[1] = DoFTools::always;
133  m[2] = DoFTools::nonzero;
134 
135  if (coupling_table.size() == nc)
136  for (unsigned int i=0; i<nc; ++i)
137  {
138  AssertThrow(coupling_table[i].size() == nc, ExcDimensionMismatch(coupling_table[i].size(), nc));
139  for (unsigned int j=0; j<nc; ++j)
140  out_coupling[i][j] = m[coupling_table[i][j]];
141  }
142  else if (coupling_table.size() == nb)
143  for (unsigned int i=0; i<nc; ++i)
144  {
145  AssertThrow(coupling_table[component_blocks[i]].size() == nb,
146  ExcDimensionMismatch(coupling_table[component_blocks[i]].size(), nb));
147  for (unsigned int j=0; j<nc; ++j)
148  out_coupling[i][j] = m[coupling_table[component_blocks[i]][component_blocks[j]]];
149  }
150  else if (coupling_table.size() == 0)
151  for (unsigned int i=0; i<nc; ++i)
152  {
153  for (unsigned int j=0; j<nc; ++j)
154  out_coupling[i][j] = m[1];
155  }
156  else
157  AssertThrow(false, ExcMessage("You tried to construct a coupling with the wrong number of elements."));
158 
159  return out_coupling;
160 }
