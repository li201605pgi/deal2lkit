89 template <int dim, int spacedim>
90 void ParsedDataOut<dim,spacedim>::parse_parameters_call_back()
91 {
92  if ( incremental_run_prefix != "" )
93  {
94  path_solution_dir = get_next_available_directory_name(incremental_run_prefix);
95  // The use of the barrier is
96  // to avoid the case of a processor below the master node.
97 #ifdef DEAL_II_WITH_MPI
98  MPI_Barrier(comm);
99 #endif
100  if ( Utilities::MPI::this_mpi_process(MPI_COMM_WORLD) == 0)
101  create_directory(path_solution_dir);
102 #ifdef DEAL_II_WITH_MPI
103  MPI_Barrier(comm);
104 #endif
105  path_solution_dir += "/";
106  }
107  else
108  {
109  path_solution_dir = "./";
110  }
111 }

113 template <int dim, int spacedim>
114 void ParsedDataOut<dim,spacedim>::prepare_data_output(const DoFHandler<dim,spacedim> &dh,
115  const std::string &suffix)
116 {
117  AssertThrow(initialized, ExcNotInitialized());
118  deallog.push("PrepareOutput");
119 
120  current_name = path_solution_dir + base_name + suffix;
121 
122  std::string fname = current_name;
123 
124  if (data_out.default_suffix() != "")
125  {
126 
127  // If the output is needed and we have many processes, just output
128  // the one we need *in intermediate format*.
129  if (n_mpi_processes > 1)
130  {
131  fname += ("." + Utilities::int_to_string(this_mpi_process, 2) +
132  "." + Utilities::int_to_string(n_mpi_processes, 2) +
133  data_out.default_suffix()) ;
134  }
135  else
136  {
137  fname += data_out.default_suffix();
138  }
139 
140  deallog << "Will write on file: " << fname.c_str() << std::endl;
141  output_file.open(fname.c_str());
142  AssertThrow(output_file, ExcIO());
143  data_out.attach_dof_handler (dh);
144 
145  if (n_mpi_processes > 1)
146  {
147  // Output the partitioning
148  if (output_partitioning)
149  {
150  deallog << "Writing partitioning" << std::endl;
151  Vector<float> partitioning(dh.get_tria().n_active_cells());
152  for (unsigned int i=0; i<partitioning.size(); ++i)
153  partitioning(i) = this_mpi_process;
154  static Vector<float> static_partitioning;
155  static_partitioning.swap(partitioning);
156  data_out.add_data_vector (static_partitioning, "partitioning");
157  }
158  }
159  }
160 
161  deallog.pop();
162 }

166 template <int dim, int spacedim>
167 void ParsedDataOut<dim,spacedim>::write_data_and_clear( const std::string &used_files,
168  const Mapping<dim,spacedim> &mapping)
169 {
170  copy_files(used_files, path_solution_dir);
171 
172  AssertThrow(initialized, ExcNotInitialized());
173  AssertThrow(output_file, ExcIO());
174  deallog.push("WritingData");
175  if (data_out.default_suffix() != "")
176  {
177  data_out.build_patches(mapping, 0,
178  DataOut<dim, DoFHandler<dim,spacedim> >::curved_inner_cells);
179  data_out.write(output_file);
180  deallog << "Wrote output file." << std::endl;
181 
182  if (this_mpi_process == 0 && n_mpi_processes > 0 && data_out.default_suffix() == ".vtu")
183  {
184  std::vector<std::string> filenames;
185  for (unsigned int i=0; i<n_mpi_processes; ++i)
186  filenames.push_back (current_name +
187  "." + Utilities::int_to_string(i, 2) +
188  "." + Utilities::int_to_string(n_mpi_processes, 2) +
189  data_out.default_suffix());
190 
191  std::ofstream master_output ((current_name + ".pvtu").c_str());
192  data_out.write_pvtu_record (master_output, filenames);
193  }
194 
195  data_out.clear();
196  output_file.close();
197  deallog << "Reset output." << std::endl;
198  }
199  deallog.pop();
200 }
